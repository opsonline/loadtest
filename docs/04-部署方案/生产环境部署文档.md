## 🔧 环境搭建

### 1. 安装Docker和Docker Compose

```bash
# 安装Docker
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh

# 安装Docker Compose
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# 添加当前用户到docker组
sudo usermod -aG docker $USER
```

### 2. 系统配置优化

```bash
# 内核参数优化
sudo bash -c 'cat >> /etc/sysctl.conf << EOF
# 文件描述符
fs.file-max = 1000000
# TCP优化
net.core.somaxconn = 65535
net.ipv4.tcp_max_syn_backlog = 65535
net.ipv4.tcp_fin_timeout = 30
# 内存优化
vm.swappiness = 10
EOF'

# 应用配置
sudo sysctl -p

# Docker守护进程配置
sudo mkdir -p /etc/docker
sudo bash -c 'cat > /etc/docker/daemon.json << EOF
{
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "100m",
    "max-file": "3"
  },
  "storage-driver": "overlay2",
  "registry-mirrors": [
    "https://mirror.aliyuncs.com"
  ]
}
EOF'

# 重启Docker服务
sudo systemctl restart docker
```

## 💾 项目部署

### 1. 拉取代码

```bash
# 创建项目目录
sudo mkdir -p /opt/loadtest
sudo chown $USER:$USER /opt/loadtest
cd /opt/loadtest

# 克隆项目代码
git clone https://github.com/your-organization/loadtest2.git .
```

### 2. 环境变量配置

创建 `.env` 文件：

```bash
sudo bash -c 'cat > .env << EOF
# 基础配置
DEBUG=False
SECRET_KEY=your-very-long-secret-key-here-replace-it-with-a-secure-random-string

# 数据库配置
DATABASE_URL=postgresql://loadtest_user:loadtest_password@db:5432/loadtest
DB_HOST=db
DB_PORT=5432
DB_NAME=loadtest
DB_USER=loadtest_user
DB_PASSWORD=loadtest_password

# Redis配置
REDIS_URL=redis://redis:6379/0

# Celery配置
CELERY_BROKER_URL=redis://redis:6379/0
CELERY_RESULT_BACKEND=redis://redis:6379/0

# 邮件配置（可选）
EMAIL_HOST=smtp.your-email.com
EMAIL_PORT=587
EMAIL_HOST_USER=your-email@your-domain.com
EMAIL_HOST_PASSWORD=your-email-password

# 日志配置
LOG_LEVEL=INFO
EOF'
```

### 3. SSL证书配置（可选）

如果需要HTTPS支持，准备SSL证书：

```bash
# 创建证书目录
sudo mkdir -p /etc/ssl/loadtest

# 将SSL证书文件放置到相应位置
# cert.pem - 证书文件
# key.pem - 私钥文件
# ca.pem - CA证书（如果有）

# 设置证书权限
sudo chmod 600 /etc/ssl/loadtest/*
sudo chown root:root /etc/ssl/loadtest/*
```

### 4. 启动服务

```bash
# 使用部署脚本（推荐）
./deploy.sh prod

# 或手动执行以下步骤：
# 1. 构建镜像
docker-compose -f docker-compose.yml build

# 2. 启动服务
docker-compose -f docker-compose.yml up -d

# 3. 等待服务启动
sleep 30

# 4. 执行数据库迁移
docker-compose -f docker-compose.yml exec backend python manage.py migrate --noinput

# 5. 收集静态文件
docker-compose -f docker-compose.yml exec backend python manage.py collectstatic --noinput
```

## ⚙️ 配置管理

### 1. Nginx配置

修改 `frontend/nginx.conf` 以适配生产环境：

```nginx
events {
    worker_connections 1024;
}

http {
    # 负载均衡配置
    upstream backend_cluster {
        server backend1:8000 weight=1 max_fails=2 fail_timeout=10s;
        server backend2:8000 weight=1 max_fails=2 fail_timeout=10s;
    }

    # 主服务器配置
    server {
        listen 80;
        server_name your-domain.com;
        
        # 重定向HTTP到HTTPS（推荐）
        return 301 https://$server_name$request_uri;
    }

    # HTTPS配置（如果启用SSL）
    server {
        listen 443 ssl http2;
        server_name your-domain.com;
        
        # SSL证书
        ssl_certificate /etc/ssl/loadtest/cert.pem;
        ssl_certificate_key /etc/ssl/loadtest/key.pem;
        ssl_trusted_certificate /etc/ssl/loadtest/ca.pem;
        
        # SSL安全配置
        ssl_protocols TLSv1.2 TLSv1.3;
        ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES256-GCM-SHA384;
        ssl_prefer_server_ciphers off;
        ssl_session_cache shared:SSL:10m;
        ssl_session_timeout 10m;
        
        # 安全头部
        add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload";
        add_header X-Frame-Options DENY;
        add_header X-Content-Type-Options nosniff;
        
        # 前端静态文件
        location / {
            root /usr/share/nginx/html;
            index index.html index.htm;
            try_files $uri $uri/ /index.html;
            
            # 缓存配置
            expires 1h;
            add_header Cache-Control "public, no-transform";
        }

        # API 请求代理到后端
        location /api/ {
            proxy_pass http://backend_cluster;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            
            # 超时配置
            proxy_connect_timeout 30s;
            proxy_send_timeout 30s;
            proxy_read_timeout 30s;
        }

        # WebSocket 代理
        location /ws/ {
            proxy_pass http://backend_cluster;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # 静态文件缓存
        location /static/ {
            alias /usr/share/nginx/html/static/;
            expires 30d;
            add_header Cache-Control "public, immutable";
        }

        # 上传文件
        location /media/ {
            alias /usr/share/nginx/html/media/;
            expires 7d;
        }
    }

    # 健康检查端点
    server {
        listen 127.0.0.1:8080;
        location /health {
            access_log off;
            return 200 "healthy\n";
        }
    }
}
```

## 🔐 安全配置

### 1. 安全加固

```bash
# 防火墙配置
sudo ufw enable
sudo ufw default deny incoming
sudo ufw default allow outgoing
sudo ufw allow ssh
sudo ufw allow 'Nginx Full'  # 或 443/tcp, 80/tcp

# SSH安全配置
sudo sed -i 's/#Port 22/Port 2222/' /etc/ssh/sshd_config  # 更改SSH端口
sudo sed -i 's/#PermitRootLogin yes/PermitRootLogin no/' /etc/ssh/sshd_config  # 禁止root登录
sudo sed -i 's/#PasswordAuthentication yes/PasswordAuthentication no/' /etc/ssh/sshd_config  # 禁用密码认证
sudo systemctl restart ssh
```

### 2. 应用安全配置

```python
# backend/config/settings.py - 安全配置
SECURE_SSL_REDIRECT = True

    'https://your-domain.com',
```

## 🔁 滚动更新

### 零停机部署脚本

```bash
#!/bin/bash
# rolling-update.sh

set -e

echo "开始滚动更新..."

# 检查当前运行的服务
docker-compose -f docker-compose.yml ps

# 拉取最新镜像
docker-compose -f docker-compose.yml pull

# 更新后端服务（蓝绿部署）
docker-compose -f docker-compose.yml up -d --no-deps --scale backend=2 backend

# 等待新实例就绪
sleep 30

# 运行健康检查
if curl -f http://localhost/api/v1/health/; then
    echo "新实例健康检查通过"
    # 停止旧实例
    docker-compose -f docker-compose.yml up -d --no-deps --scale backend=1 backend
else
    echo "新实例健康检查失败，回滚..."
    docker-compose -f docker-compose.yml up -d --no-deps --scale backend=1 backend
    exit 1
fi

echo "滚动更新完成！"
```
