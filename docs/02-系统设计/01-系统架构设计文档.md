# 系统架构设计文档

## 一、系统概述

### 1.1 系统架构

本系统采用前后端分离的 B/S 架构，基于 Django + Vue.js 构建。系统整体分为四个层次：表现层、业务逻辑层、数据访问层和数据存储层。

系统架构采用微服务思想设计，各功能模块独立部署，通过 RESTful API 进行通信。压测执行引擎独立运行，通过消息队列与主系统异步通信。

### 1.2 系统拓扑

整体系统由以下组件构成：

| 组件 | 职责 | 部署方式 |
|------|------|----------|
| Web 前端 | 提供用户界面 | Nginx 静态部署 |
| API 服务器 | 处理业务逻辑 | Gunicorn 多进程 |
| 消息队列 | 异步任务调度 | Redis |
| 压测引擎 | 执行压测任务 | Docker 容器 |
| 文件存储 | 存储上传文件 | 本地文件系统 |
| 数据库 | 持久化存储 | SQLite（开发）/MySQL（生产） |

#### 3.1.1 组件通信模式

| 模式 | 适用场景 | 实现方式 |
|------|----------|----------|
| Props/Emits | 父子组件通信 | Vue 响应式 props、emit 事件 |
| Provide/Inject | 跨层级组件通信 | Vue 注入机制 |
| Pinia Store | 全局状态管理 | getter、action、state |
| EventBus | 跨组件事件 | mitt 或 Vue 实例 |

```python
# services/base_service.py
from django.db import transaction
from typing import TypeVar, Generic, List, Optional
from django.db.models import Model

T = TypeVar('T', bound=Model)

class BaseService(Generic[T]):
    model_class: Type[T] = None
    
    def get_queryset(self):
        return self.model_class.objects.all()
    
    def get(self, pk) -> Optional[T]:
        try:
            return self.get_queryset().get(pk=pk)
        except self.model_class.DoesNotExist:
            return None
    
    def list(self, filters=None, ordering=None, pagination=None):
        queryset = self.get_queryset()
        
        if filters:
            queryset = queryset.filter(**filters)
        
        if ordering:
            queryset = queryset.order_by(*ordering)
        else:
            queryset = queryset.order_by('-created_at')
        
        if pagination:
            offset = pagination.get('offset', 0)
            limit = pagination.get('limit', 20)
            queryset = queryset[offset:offset + limit]
        
        return queryset
    
    @transaction.atomic
    def create(self, data: dict) -> T:
        return self.model_class.objects.create(**data)
    
    @transaction.atomic
    def update(self, instance: T, data: dict) -> T:
        for key, value in data.items():
            setattr(instance, key, value)
        instance.save()
        return instance
    
    @transaction.atomic
    def delete(self, instance: T) -> bool:
        instance.delete()
        return True

# services/scenario_service.py
from .base_service import BaseService
from har_convert.models import Scenario, Request

class ScenarioService(BaseService):
    model_class = Scenario
    
    def get_with_requests(self, scenario_id):
        return self.get_queryset().prefetch_related('requests').get(pk=scenario_id)
    
    @transaction.atomic
    def create_with_requests(self, scenario_data, requests_data):
        scenario = self.create(scenario_data)
        for idx, req_data in enumerate(requests_data):
            Request.objects.create(scenario=scenario, order=idx, **req_data)
        return scenario
    
    @transaction.atomic
    def duplicate(self, scenario_id):
        original = self.get_with_requests(scenario_id)
        requests_data = []
        for req in original.requests.all():
            req_data = {
                'name': req.name,
                'method': req.method,
                'url': req.url,
                'headers': req.headers,
                'body': req.body,
                'body_type': req.body_type,
                'weight': req.weight,
                'think_time': req.think_time,
                'timeout': req.timeout,
                'is_enabled': req.is_enabled
            }
            requests_data.append(req_data)
        
        return self.create_with_requests({
            'name': f'{original.name} (副本)',
            'description': original.description,
            'target_host': original.target_host,
            'config': original.config
        }, requests_data)
```

```python
@app.task(bind=True, max_retries=3, default_retry_delay=60)
```

```python
# consumers/run_consumer.py
import json
from channels.generic.websocket import AsyncWebsocketConsumer
from channels.db import database_sync_to_async
import redis.asyncio as aioredis

class RunStatsConsumer(AsyncWebsocketConsumer):
    async def connect(self):
        self.run_id = self.scope['url_route']['kwargs']['run_id']
        self.user = self.scope['user']
        
        # 验证用户权限
        if not await self.has_permission():
            await self.close(code=4001)
            return
        
        self.group_name = f'run_{self.run_id}'
        await self.channel_layer.group_add(self.group_name, self.channel_name)
        await self.accept()
        
        # 启动数据推送
        await self.start_stats_push()
    
    async def disconnect(self, close_code):
        await self.channel_layer.group_discard(self.group_name, self.channel_name)
    
    async def receive(self, text_data):
        data = json.loads(text_data)
        
        if data.get('type') == 'ping':
            await self.send(text_data=json.dumps({'type': 'pong'}))
        elif data.get('type') == 'control':
            await self.handle_control(data)
    
    async def handle_control(self, data):
        if data.get('action') == 'stop':
            await stop_load_test.delay(self.run_id)
    
    async def send_stats(self, event):
        await self.send(text_data=json.dumps(event['data']))
    
    async def start_stats_push(self):
        # 从 Redis 获取统计数据并推送
        redis = await aioredis.from_url('redis://localhost:6379')
        
        while True:
            stats = await redis.hgetall(f'run:{self.run_id}:stats')
            if stats:
                await self.send(text_data=json.dumps({
                    'type': 'stats',
                    'data': stats
                }))
            await asyncio.sleep(1)
    
    @database_sync_to_async
    def has_permission(self):
        from har_convert.models import Run
        try:
            run = Run.objects.get(id=self.run_id)
            return run.created_by == self.user
        except Run.DoesNotExist:
            return False
```

### 4.2 轮询备选方案

```javascript
// utils/polling.js
class PollingManager {
  constructor(options = {}) {
    this.fetchFn = options.fetchFn
    this.interval = options.interval || 1000
    this.handlers = new Set()
    this.timer = null
    this.isRunning = false
  }
  
  start() {
    if (this.isRunning) return
    
    this.isRunning = true
    this.timer = setInterval(async () => {
      try {
        const data = await this.fetchFn()
        this.handlers.forEach(handler => handler(data))
      } catch (error) {
        console.error('Polling error:', error)
      }
    }, this.interval)
  }
  
  stop() {
    if (this.timer) {
      clearInterval(this.timer)
      this.timer = null
    }
    this.isRunning = false
  }
  
  onUpdate(handler) {
    this.handlers.add(handler)
    return () => this.handlers.delete(handler)
  }
}

// 使用示例
const statsPolling = new PollingManager({
  interval: 2000,
  fetchFn: () => axios.get(`/api/runs/${runId}/stats/`).then(r => r.data)
})

const unsubscribe = statsPolling.onUpdate((stats) => {
  updateUI(stats)
})

// 组件卸载时停止轮询
onUnmounted(() => {
  statsPolling.stop()
  unsubscribe()
})
```

### 5.1 认证授权

#### 5.1.1 JWT 认证流程

```python
# users/authentication.py
import jwt
from datetime import datetime, timedelta
from django.conf import settings
from rest_framework.authentication import BaseAuthentication
from rest_framework.exceptions import AuthenticationFailed
from .models import User

class JWTAuthentication(BaseAuthentication):
    keyword = 'Bearer'
    
    def authenticate(self, request):
        auth_header = request.META.get('HTTP_AUTHORIZATION', '')
        
        if not auth_header.startswith(self.keyword):
            return None
        
        token = auth_header[len(self.keyword) + 1:]
        
        try:
            payload = jwt.decode(
                token,
                settings.JWT_SECRET_KEY,
                algorithms=['HS256']
            )
        except jwt.ExpiredSignatureError:
            raise AuthenticationFailed('Token 已过期')
        except jwt.InvalidTokenError:
            raise AuthenticationFailed('无效的 Token')
        
        try:
            user = User.objects.get(id=payload['user_id'])
        except User.DoesNotExist:
            raise AuthenticationFailed('用户不存在')
        
        if not user.is_active:
            raise AuthenticationFailed('用户已被禁用')
        
        return (user, token)
    
    def authenticate_header(self, request):
        return self.keyword

# users/tokens.py
def generate_tokens(user):
```

```python
from cryptography.fernet import Fernet

class DataEncryptor:
    def __init__(self):
        self.fernet = Fernet(settings.ENCRYPTION_KEY.encode())
```

```python
# utils/metrics.py
import time
from functools import wraps

def track_performance(operation_name):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            start_time = time.time()
            result = func(*args, **kwargs)
            duration = (time.time() - start_time) * 1000
            
            # 记录到指标系统
            record_metric('api_duration', duration, {'operation': operation_name})
            
            return result
        return wrapper
    return decorator

def record_metric(name, value, tags=None):
    # 可以接入 Prometheus、InfluxDB 等监控系统
    print(f'[METRIC] {name}={value} tags={tags}')
```
